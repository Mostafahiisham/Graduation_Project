from setuptools import setup

package_name = 'motor_encoder_ros2'

setup(
    name=package_name,
    version='0.0.0',
    packages=[package_name],
    data_files=[
        ('share/ament_index/resource_index/packages',
            ['resource/' + package_name]),
        ('share/' + package_name, ['package.xml']),
    ],
    install_requires=['setuptools'],
    zip_safe=True,
    maintainer='your_name',
    maintainer_email='your_email@example.com',
    description='ROS2 package for motor encoder monitoring',
    license='Apache License 2.0',
    tests_require=['pytest'],
    entry_points={
        'console_scripts': [
            'encoder_publisher = motor_encoder_ros2.encoder_publisher:main',
            'encoder_subscriber = motor_encoder_ros2.encoder_subscriber:main',
        ],
    },
)




# In your workspace root
colcon build --packages-select motor_encoder_ros2
source install/setup.bash

# Start pigpio daemon if not already running
sudo pigpiod

ros2 run motor_encoder_ros2 encoder_publisher

# In another terminal
ros2 run motor_encoder_ros2 encoder_subscriber
